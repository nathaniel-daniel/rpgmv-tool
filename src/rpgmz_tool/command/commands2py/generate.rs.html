13pub fn commands2py<W>(
14    config: &Config,
15    commands: &[(u16, Command)],
16    mut writer: W,
17) -> anyhow::Result<()>
18where
19    W: Write,
20{
21    for (indent, command) in commands.iter() {
22        command2py(config, *indent, command, &mut writer)?;
23    }
24
25    Ok(())
26}

28fn command2py<W>(
29    config: &Config,
30    indent: u16,
31    command: &Command,
32    mut writer: W,
33) -> anyhow::Result<()>
34where
35    W: Write,
36{
37    match command {
38        Command::Nop => {}
39        Command::ShowText {
40            face_name,
41            face_index,
42            background,
43            position_type,
44            speaker_name,
45            lines,
46        } => {
47            let mut writer = FunctionCallWriter::new(&mut writer, indent, "show_text")?;
48            writer.write_param("face_name", face_name)?;
49            writer.write_param("face_index", face_index)?;
50            writer.write_param("background", background)?;
51            writer.write_param("position_type", position_type)?;
52            writer.write_param("speaker_name", speaker_name)?;
53            writer.write_param("lines", lines)?;
54            writer.finish()?;
55        }
56        Command::ShowChoices {
57            choices,
58            cancel_type,
59            default_type,
60            position_type,
61            background,
62        } => {
63            let mut writer = FunctionCallWriter::new(&mut writer, indent, "show_choices")?;
64            writer.write_param("choices", choices)?;
65            writer.write_param("cancel_type", cancel_type)?;
66            writer.write_param("default_type", default_type)?;
67            writer.write_param("position_type", position_type)?;
68            writer.write_param("background", background)?;
69            writer.finish()?;
70        }
71        Command::Comment { lines } => {
72            for line in lines.iter() {
73                write_indent(&mut writer, indent)?;
74                writeln!(&mut writer, "# {line}")?;
75            }
76        }
77        Command::ConditionalBranch(command) => {
78            write_indent(&mut writer, indent)?;
79            write!(&mut writer, "if ")?;
80            match command {
81                ConditionalBranchCommand::Switch { id, check_true } => {
82                    let name = config.get_switch_name(*id);
83                    let check_true_str = if *check_true { "" } else { "not " };
84                    writeln!(&mut writer, "{check_true_str}{name}:")?;
85                }
86                ConditionalBranchCommand::Variable {
87                    lhs_id,
88                    rhs_id,
89                    operation,
90                } => {
91                    let lhs = config.get_variable_name(*lhs_id);
92                    let rhs = match rhs_id {
93                        MaybeRef::Constant(value) => value.to_string(),
94                        MaybeRef::Ref(id) => config.get_variable_name(*id),
95                    };
96                    let operation = operation.as_str();
97
98                    writeln!(&mut writer, "{lhs} {operation} {rhs}:")?;
99                }
100                ConditionalBranchCommand::SelfSwitch { name, check_true } => {
101                    let name = escape_string(name);
102                    let check_true_str = if *check_true { "" } else { "not " };
103                    writeln!(&mut writer, "{check_true_str}game_self_switches.get(map_id=self.map_id, event_id=self.event_id, name='{name}'):")?;
104                }
105                ConditionalBranchCommand::ActorInParty { actor_id } => {
106                    let actor_name = config.get_actor_name(*actor_id);
107
108                    writeln!(
109                        &mut writer,
110                        "game_party.members.contains(actor={actor_name}):"
111                    )?;
112                }
113                ConditionalBranchCommand::Timer { value, is_gte } => {
114                    let cmp = if *is_gte { ">=" } else { "<=" };
115
116                    writeln!(&mut writer, "game_timer.seconds() {cmp} {value}:")?;
117                }
118                ConditionalBranchCommand::ActorSkill { actor_id, skill_id } => {
119                    let actor_name = config.get_actor_name(*actor_id);
120                    let skill_name = config.get_skill_name(*skill_id);
121
122                    writeln!(&mut writer, "{actor_name}.has_skill(skill={skill_name}):")?;
123                }
124                ConditionalBranchCommand::ActorArmor { actor_id, armor_id } => {
125                    let actor_name = config.get_actor_name(*actor_id);
126                    let armor_name = config.get_armor_name(*armor_id);
127
128                    writeln!(&mut writer, "{actor_name}.has_armor(armor={armor_name}):")?;
129                }
130                ConditionalBranchCommand::ActorState { actor_id, state_id } => {
131                    let actor_name = config.get_actor_name(*actor_id);
132                    let state_name = config.get_state_name(*state_id);
133
134                    writeln!(&mut writer, "{actor_name}.has_state(state={state_name}):")?;
135                }
136                ConditionalBranchCommand::EnemyState {
137                    enemy_index,
138                    state_id,
139                } => {
140                    let name = config.get_state_name(*state_id);
141
142                    writeln!(
143                        &mut writer,
144                        "game_troop.members[{enemy_index}].is_state_affected(state={name}):"
145                    )?;
146                }
147                ConditionalBranchCommand::Character {
148                    character_id,
149                    direction,
150                } => {
151                    let name = if *character_id < 0 {
152                        "game_player".to_string()
153                    } else {
154                        format!("game_character_{character_id}")
155                    };
156
157                    writeln!(&mut writer, "{name}.direction == {direction}:")?;
158                }
159                ConditionalBranchCommand::Gold { value, check } => {
160                    let check = check.as_str();
161
162                    writeln!(&mut writer, "game_party.gold {check} {value}:")?;
163                }
164                ConditionalBranchCommand::Item { item_id } => {
165                    let name = config.get_item_name(*item_id);
166
167                    writeln!(&mut writer, "game_party.has_item(item={name}):")?;
168                }
169                ConditionalBranchCommand::Button { key_name } => {
170                    let key_name = escape_string(key_name);
171
172                    writeln!(&mut writer, "game_input.is_pressed(key_name='{key_name}'):")?;
173                }
174                ConditionalBranchCommand::Script { value } => {
175                    let value = escape_string(value);
176
177                    writeln!(&mut writer, "execute_script('{value}'):")?;
178                }
179            }
180        }
181        Command::Loop => {
182            write_indent(&mut writer, indent)?;
183            writeln!(&mut writer, "while True:")?;
184        }
185        Command::CommonEvent { id } => {
186            let name = config.get_common_event_name(*id);
187            FunctionCallWriter::new(&mut writer, indent, &name)?.finish()?;
188        }
189        Command::Label { name } => {
190            let mut writer = FunctionCallWriter::new(&mut writer, indent, "set_label")?;
191            writer.set_multiline(false);
192            writer.write_param("name", name)?;
193            writer.finish()?;
194        }

222                        format!("{name}.level")
223                    }

225                        let name = config.get_actor_name(*actor_id);
226                        format!("{name}.hp")
227                    }
228                    ControlVariablesValueGameData::ActorMp { actor_id } => {
229                        let name = config.get_actor_name(*actor_id);
230                        format!("{name}.mp")
231                    }
232                    ControlVariablesValueGameData::ActorParam { param_index } => {
233                        format!("game_param_{param_index}")
234                    }
235                    ControlVariablesValueGameData::CharacterMapX { character_id } => {
236                        format!("game.get_character(id={character_id}).map_x")
237                    }
238                    ControlVariablesValueGameData::CharacterMapY { character_id } => {
239                        format!("game.get_character(id={character_id}).map_y")
240                    }
241                    ControlVariablesValueGameData::CharacterScreenX { character_id } => {
242                        format!("game.get_character(id={character_id}).screen_x")
243                    }
244                    ControlVariablesValueGameData::CharacterScreenY { character_id } => {
245                        format!("game.get_character(id={character_id}).screen_y")
246                    }
247                    ControlVariablesValueGameData::MapId => "game_map.map_id()".to_string(),
248                    ControlVariablesValueGameData::Gold => "game_party.gold".to_string(),
249                    ControlVariablesValueGameData::Steps => "game_party.steps".to_string(),
250                },
251                ControlVariablesValue::Script { value } => {
252                    let value = escape_string(value);
253                    format!("execute_script('{value}')")
254                }
255            };
256            for variable_id in *start_variable_id..(*end_variable_id + 1) {
257                let name = config.get_variable_name(variable_id);
258
259                write_indent(&mut writer, indent)?;
260                writeln!(&mut writer, "{name} {operation} {value}")?;
261            }
262        }
263        Command::ControlSwitches {
264            start_id,
265            end_id,
266            value,
267        } => {
268            for id in *start_id..(*end_id + 1) {
269                let name = config.get_switch_name(id);
270                let value = stringify_bool(*value);

272                write_indent(&mut writer, indent)?;
273                writeln!(&mut writer, "{name} = {value}")?;

275        }

664    Ok(())
665}

674fn write_indent<W>(mut writer: W, indent: u16) -> std::io::Result<()>
675where
676    W: Write,
677{
678    for _ in 0..indent {
679        write!(writer, "\t")?;
680    }
681
682    Ok(())
683}
