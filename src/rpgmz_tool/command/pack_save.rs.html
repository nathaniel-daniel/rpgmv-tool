9#[derive(Debug, argh::FromArgs)]
10#[argh(subcommand, name = "pack-save", description = "a tool to pack a save")]
11pub struct Options {
12    #[argh(positional)]
13    pub input: PathBuf,
14
15    #[argh(positional)]
16    pub output: PathBuf,
17}
18
19fn prepare_compress_output(input: &[u8]) -> String {
20    let mut output = String::new();
21    for value in input.iter().copied() {
22        output.push(char::from(value));
23    }
24    output
25}
26
27pub fn exec(options: Options) -> anyhow::Result<()> {
28    let input = std::fs::read_to_string(&options.input)
29        .with_context(|| format!("failed to read file at \"{}\"", options.input.display()))?;
30    let input: serde_json::Value =
31        serde_json::from_str(&input).context("save file is not valid json")?;
32    let input = serde_json::to_string(&input)?;
33
34    let mut encoder = ZlibEncoder::new(std::io::Cursor::new(input), Compression::fast());
35    let mut output = Vec::new();
36    encoder.read_to_end(&mut output)?;
37
38    let output = prepare_compress_output(&output);
39
40    let tmp_out_path = nd_util::with_push_extension(&options.output, "tmp");
41    let mut out_file = File::create(&tmp_out_path)
42        .with_context(|| format!("failed to create file at \"{}\"", tmp_out_path.display()))?;
43    out_file.write_all(output.as_bytes())?;
44    out_file.flush()?;
45    out_file.sync_all()?;
46
47    std::fs::rename(&tmp_out_path, &options.output).context("failed to rename output file")?;
48
49    Ok(())
50}
