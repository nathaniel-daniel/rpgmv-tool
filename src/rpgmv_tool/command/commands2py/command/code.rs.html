1macro_rules! command_codes {
2    (
3        $(
4            $(#[$docs:meta])*
5            ($id:expr, $name:ident),
6        )+
7    ) => {
8        impl CommandCode {
9            $(
10                $(#[$docs])*
11                pub const $name: Self = Self($id);
12            )+
13
14            pub fn as_str(self) -> Option<&'static str> {
15                match self {
16                    $(
17                    Self::$name => Some(stringify!($name)),
18                    )+
19                    _ => None
20                }
21            }
22        }
23    }
24}

171impl std::fmt::Debug for CommandCode {
172    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
173        match self.as_str() {
174            Some(s) => write!(f, "{s}"),
175            None => write!(f, "Unknown({})", self.0),
176        }
177    }
