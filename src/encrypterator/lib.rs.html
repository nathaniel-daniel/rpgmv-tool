27/// Prepare the encryption key.
28///
29/// This will always return a `Vec<u8>` with length 32.
30fn transform_encryption_key(input: &str) -> [u8; TRANSFORMED_KEY_LEN] {
31    let mut hasher = Md5::new();
32    hasher.update(input.as_bytes());
33    let hash = hasher.finalize();
34
35    let mut output = hash.to_vec();
36    output.extend(hash.iter().rev());
37
38    *output.as_array().expect("invalid key len")
39}
