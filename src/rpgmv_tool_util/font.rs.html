29/// Load a font.
30pub fn load_font(path: &Path) -> anyhow::Result<Font> {
31    let buffer = std::fs::read(path)?;
32    let font = FontTryBuilder {
33        buffer,
34        scope_builder: |buffer| Ok(ReadScope::new(buffer)),
35        font_data_builder: |scope| {
36            scope
37                .read::<FontData<'_>>()
38                .context("failed to read font data")
39        },
40        font_builder: |font_file| {
41            let table_provider = font_file
42                .table_provider(0)
43                .context("unable to create table provider")?;
44            allsorts::Font::new(table_provider).context("unable to load font tables")
45        },
46    }
47    .try_build()?;
48
49    Ok(font)
50}

52/// Estimate the width of some text.
53pub fn get_text_width(font: &mut Font, text: &str, font_size: Option<f32>) -> anyhow::Result<f32> {
54    let script = tag::LATN;
55    let lang = tag::DFLT;
56    let variation_tuple = None;
57
58    let text_size = font.with_font_mut(|font| {
59        let scale = font_size.map(|font_size| font_size / f32::from(font.head_table.units_per_em));
60
61        let glyphs = font.map_glyphs(text, script, MatchingPresentation::NotRequired);
62
63        let glyph_infos = font
64            .shape(
65                glyphs,
66                script,
67                Some(lang),
68                FeatureMask::default_mask(),
69                &[],
70                variation_tuple,
71                true,
72            )
73            .map_err(|(err, _)| err)
74            .context("failed to shape text")?;
75
76        let mut layout = GlyphLayout::new(font, &glyph_infos, TextDirection::LeftToRight, false);
77        let positions = layout.glyph_positions()?;
78
79        let mut text_width_unscaled: i32 = 0;
80        for (_glyph, position) in glyph_infos.iter().zip(&positions) {
81            text_width_unscaled += position.hori_advance;
82        }
83        let text_width_scaled = match scale {
84            Some(scale) => (text_width_unscaled as f32) * scale,
85            None => text_width_unscaled as f32,
86        };
87        anyhow::Ok(text_width_scaled)
88    })?;
89
90    Ok(text_size)
91}
