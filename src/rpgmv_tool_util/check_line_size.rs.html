18/// Message text is padded by default with 18px.
19const MESSAGE_STANDARD_PADDING: u16 = 18;
20
21/// Message Actor Faces are 144x144 px.
22const MESSAGE_FACE_SIZE: u16 = 144;
23const MESSAGE_FACE_PADDING: u16 = 12;
24/// Text padding is applied in addition to STANDARD_PADDING, but not for all windows.
25const TEXT_PADDING: u16 = 6;

27fn read_to_string<P>(path: P) -> anyhow::Result<String>
28where
29    P: AsRef<Path>,
30{
31    let path = path.as_ref();
32    std::fs::read_to_string(path).with_context(|| format!("Failed to read \"{}\"", path.display()))
33}

35/// MV only
36fn load_plugins_js(game_path: &Path) -> anyhow::Result<Vec<Plugin>> {
37    static REGEX: LazyLock<Regex> =
38        LazyLock::new(|| Regex::new(r"(?s)var \$plugins =\s*(\[.*\]);").unwrap());
39
40    // TODO: If we ever add MZ support, change the path here conditonally.
41    let js_dir = game_path.join("www").join("js");
42    let plugins_js_path = js_dir.join("plugins.js");
43    let plugins_js_raw = std::fs::read_to_string(plugins_js_path)?;
44    let plugins_js_string = REGEX
45        .captures(&plugins_js_raw)
46        .context("Failed to locate plugins with regex")?
47        .get(1)
48        .context("Missing group")?
49        .as_str();
50    let plugins = serde_json::from_str(plugins_js_string)?;
51    Ok(plugins)
52}

104    fn suggest_line_replacement(
105        &mut self,
106        line: &str,
107        target_width: u16,
108    ) -> anyhow::Result<Option<String>> {
109        if self.get_text_width(line)? < f32::from(target_width) {
110            return Ok(Some(line.to_string()));
111        }
112
113        // This can be a lot more efficient if we use raw harfbuzz/whatever text layout library we are using.
114        let mut words = line.split(' ').collect::<Vec<_>>();
115        while !words.is_empty() {
116            words.pop();
117            let candidate = words.join(" ");
118            if candidate.is_empty() {
119                return Ok(None);
120            }
121            let suggested_width = self.get_text_width(&candidate)?;
122            if suggested_width < f32::from(target_width) {
123                return Ok(Some(candidate));
124            }
125        }
126
127        Ok(None)
128    }

130    fn check_event_command_message(
131        &mut self,
132        face: &str,
133        lines: &str,
134        message_width: u16,
135        file: &str,
136    ) -> anyhow::Result<()> {
137        // Strip escape sequences.
138        let mut parser = MessageParser::new(lines).yep_message_core(self.yep_message_core);
139        if self.yep_message_core {
140            parser = parser.add_yep_message_core_text_codes();
141        }
142        for single_text_code in self.extra_single_text_codes.iter() {
143            parser.add_single_text_code(single_text_code);
144        }
145        for text_code in self.extra_text_codes.iter() {
146            parser.add_text_code(text_code);
147        }
148        let nodes = parser
149            .parse()
150            .with_context(|| format!("Failed to parse \"{lines}\""))?;
151        let mut stripped_lines = String::with_capacity(lines.len());
152        for node in nodes {
153            match node {
154                MessageNode::Text { value } => stripped_lines.push_str(&value),
155                MessageNode::TextCode { .. }
156                | MessageNode::TextCodeWithBody { .. }
157                | MessageNode::YepTextCodeWithBody { .. } => {
158                    // Ignore text codes
159                }
160            }
161        }
162
163        for line in stripped_lines.split('\n') {
164            if line.is_empty() {
165                continue;
166            }
167
168            let text_width = self.get_text_width(line)?;
169            let mut target_width = message_width - (2 * MESSAGE_STANDARD_PADDING);
170            if !face.is_empty() {
171                target_width -= MESSAGE_FACE_SIZE + (MESSAGE_FACE_PADDING * 2);
172            }
173
174            if text_width > f32::from(target_width) {
175                let suggested_line = self.suggest_line_replacement(line, target_width)?;
176                let suggested_line = suggested_line.as_deref().unwrap_or("None");
177
178                self.entries.push_back(CheckLineSizeEntry {
179                    file: file.to_string(),
180                    line: line.to_string(),
181                    text_width,
182                    target_width,
183                    suggested_line: suggested_line.to_string(),
184                });
185            }
186        }
187
188        Ok(())
189    }

243    fn check_event_command_list(
244        &mut self,
245        command_list: &[rpgmv_types::EventCommand],
246        file: &str,
247    ) -> anyhow::Result<()> {
248        const SHOW_TEXT: u32 = 101;
249        const PLUGIN_COMMAND: u32 = 356;
250        const TEXT_DATA: u32 = 401;
251
252        let mut last_face = "";
253        let mut lines = None;
254        let mut message_width = self.game_width;
255        for command in command_list.iter() {
256            let code = command.code;
257            let parameters = &command.parameters;
258
259            if code != SHOW_TEXT && code != TEXT_DATA {
260                if let Some(lines) = lines.as_deref() {
261                    self.check_event_command_message(last_face, lines, message_width, file)?;
262                }
263
264                last_face = "";
265                lines = None;
266            }
267
268            match code {
269                SHOW_TEXT => {
270                    if let Some(lines) = lines.as_deref() {
271                        self.check_event_command_message(last_face, lines, message_width, file)?;
272                    }
273
274                    last_face = parameters
275                        .first()
276                        .context("Missing last face")?
277                        .as_str()
278                        .context("Face is not a string")?;
279                    lines = Some(String::new());
280                }
281                PLUGIN_COMMAND if self.yep_message_core => {
282                    let value = parameters
283                        .first()
284                        .context("Missing plugin parameter")?
285                        .as_str()
286                        .context("Plugin parameter is not a string")?;
287                    if let Some(("MessageWidth", args)) = value.split_once(' ') {
288                        let width: u16 = args.parse()?;
289
290                        message_width = width;
291                    }
292                }
293                TEXT_DATA => {
294                    let line = parameters
295                        .first()
296                        .context("Missing line")?
297                        .as_str()
298                        .context("Line is not a string")?;
299
300                    let lines = lines
301                        .as_mut()
302                        .context("Encountered TEXT_DATA command out of message")?;
303                    if !lines.is_empty() {
304                        lines.push('\n');
305                    }
306                    lines.push_str(line);
307                }
308                _ => {}
309            }
310        }
311
312        if let Some(lines) = lines.as_deref() {
313            self.check_event_command_message(last_face, lines, message_width, file)?;
314        }
315
316        Ok(())
317    }

419pub struct CheckLineSizeIter {
420    found_error: bool,
421    dir_entries: std::vec::IntoIter<std::fs::DirEntry>,
422    context: CheckLineSizeContext,
423}
424
425impl CheckLineSizeIter {
426    fn new(path: &Path, context: CheckLineSizeContext) -> anyhow::Result<Self> {
427        let mut dir_entries = std::fs::read_dir(path)?.collect::<Result<Vec<_>, _>>()?;
428        dir_entries.sort_by_key(|entry_a| entry_a.file_name());
429
430        Ok(Self {
431            found_error: false,
432            dir_entries: dir_entries.into_iter(),
433            context,
434        })
435    }
436}
437
438impl Iterator for CheckLineSizeIter {
439    type Item = anyhow::Result<CheckLineSizeEntry>;
440
441    fn next(&mut self) -> Option<Self::Item> {
442        if self.found_error {
443            return None;
444        }
445
446        if let Some(entry) = self.context.pop_entry() {
447            return Some(Ok(entry));
448        }
449
450        loop {
451            let entry = self.dir_entries.next()?;
452
453            let result = (|| {
454                let entry_file_name = entry
455                    .file_name()
456                    .to_str()
457                    .context("file name is not unicode")?
458                    .to_string();
459
460                let entry_path = entry.path();
461
462                if !entry_file_name.ends_with(".json") {
463                    return Ok(());
464                }
465                if entry_file_name == "MapInfos.json" {
466                    return Ok(());
467                }
468
469                if let Some(map_number) = parse_map_name(&entry_file_name) {
470                    let string = read_to_string(&entry_path)?;
471                    let map: rpgmv_types::Map = serde_json::from_str(&string)
472                        .with_context(|| format!("Failed to parse Map {map_number}"))?;
473                    self.context.check_map(&map, map_number)?;
474                } else {
475                    match entry_file_name.as_str() {
476                        "CommonEvents.json" => {
477                            let string = read_to_string(&entry_path)?;
478                            let value: Vec<Option<rpgmv_types::CommonEvent>> =
479                                serde_json::from_str(&string)
480                                    .context("Failed to parse CommonEvents")?;
481                            self.context.check_common_events(&value)?;
482                        }
483                        "Troops.json" => {
484                            let string = read_to_string(&entry_path)?;
485                            let value: Vec<Option<rpgmv_types::Troop>> =
486                                serde_json::from_str(&string).context("Failed to parse Troops")?;
487                            self.context.check_troops(&value)?;
488                        }
489                        "Armors.json" => {
490                            let string = read_to_string(&entry_path)?;
491                            let value: Vec<Option<rpgmv_types::Armor>> =
492                                serde_json::from_str(&string).context("Failed to parse Armors")?;
493                            self.context.check_armors(&value)?;
494                        }
495                        "Items.json" => {
496                            let string = read_to_string(&entry_path)?;
497                            let value: Vec<Option<rpgmv_types::Item>> =
498                                serde_json::from_str(&string).context("Failed to parse Items")?;
499                            self.context.check_items(&value)?;
500                        }
501                        "Skills.json" => {
502                            let string = read_to_string(&entry_path)?;
503                            let value: Vec<Option<rpgmv_types::Skill>> =
504                                serde_json::from_str(&string).context("Failed to parse Skills")?;
505                            self.context.check_skills(&value)?;
506                        }
507                        "Weapons.json" => {
508                            let string = read_to_string(&entry_path)?;
509                            let value: Vec<Option<rpgmv_types::Weapon>> =
510                                serde_json::from_str(&string).context("Failed to parse Weapons")?;
511                            self.context.check_weapons(&value)?;
512                        }
513                        _ => {}
514                    }
515                }
516
517                anyhow::Ok(())
518            })();
519
520            match result {
521                Ok(()) => {
522                    if let Some(entry) = self.context.pop_entry() {
523                        return Some(Ok(entry));
524                    }
525                }
526                Err(error) => {
527                    self.found_error = true;
528                    return Some(Err(error));
529                }
530            }
531        }
532    }
533}

571impl CheckLineSizeOptions {
572    /// Create options for checking a game from a game path.
573    pub fn from_game_path<P>(game_path: P) -> anyhow::Result<Self>
574    where
575        P: AsRef<Path>,
576    {
577        let game_path = game_path.as_ref();
578
579        let game_is_mv = is_game_mv(game_path)?;
580
581        let mut screen_width = 816;
582        let mut yep_message_core = false;
583        let (font_name, font_size) = if game_is_mv {
584            let plugins = load_plugins_js(game_path)?;
585
586            for plugin in plugins {
587                if !plugin.status {
588                    continue;
589                }
590
591                match plugin.name.as_str() {
592                    "Community_Basic" => {
593                        let screen_width_param = plugin
594                            .parameters
595                            .get("screenWidth")
596                            .context("Missing Community_Basic screenWidth")?;
597                        let screen_width_param = if screen_width_param.is_empty() {
598                            None
599                        } else {
600                            Some(
601                                screen_width_param
602                                    .parse()
603                                    .context("Community_Basic screenWidth is not a u16")?,
604                            )
605                        };
606                        // This plugin will set a default if not present.
607                        let screen_width_param = screen_width_param.unwrap_or(816);
608                        screen_width = screen_width_param;
609                    }
610                    "YEP_CoreEngine" => {
611                        let screen_width_param = plugin
612                            .parameters
613                            .get("Screen Width")
614                            .context("Missing YEP_CoreEngine Screen Width")?;
615                        let screen_width_param = if screen_width_param.is_empty() {
616                            None
617                        } else {
618                            Some(
619                                screen_width_param
620                                    .parse()
621                                    .context("YEP_CoreEngine Screen Width is not a u16")?,
622                            )
623                        };
624                        // This plugin will set a default if not present.
625                        let screen_width_param = screen_width_param.unwrap_or(816);
626                        screen_width = screen_width_param;
627                    }
628                    "YEP_MessageCore" => {
629                        yep_message_core = true;
630                    }
631                    _ => {}
632                }
633            }
634
635            ("mplus-1m-regular.ttf".to_string(), 28)
636        } else {
637            let system_path = game_path.join("data").join("System.json");
638            let system_string = std::fs::read_to_string(&system_path).with_context(|| {
639                format!("Failed to read to string \"{}\"", system_path.display())
640            })?;
641            let system: System =
642                serde_json::from_str(&system_string).context("Failed to parse system json")?;
643            let system_advanced = system
644                .advanced
645                .context("System missing \"advanced\" field")?;
646
647            screen_width = system_advanced.screen_width;
648
649            (
650                system_advanced.main_font_filename,
651                system_advanced.font_size,
652            )
653        };
654
655        let data_path = {
656            let mut path = PathBuf::from(game_path);
657            if game_is_mv {
658                path.push("www");
659            }
660            path.push("data");
661            path
662        };
663
664        Ok(Self {
665            game_path: game_path.to_path_buf(),
666            game_is_mv,
667            font_name,
668            font_size,
669            screen_width,
670            data_path,
671            extra_single_text_codes: HashSet::new(),
672            extra_text_codes: HashSet::new(),
673            yep_message_core,
674        })
675    }
676
677    fn load_font(&self) -> anyhow::Result<Font> {
678        let font_path = {
679            let mut font_path = PathBuf::from(&self.game_path);
680            if self.game_is_mv {
681                font_path.push("www");
682            }
683            font_path.push("fonts");
684            font_path.push(&self.font_name);
685            font_path
686        };
687        let font = load_font(&font_path)?;
688
689        Ok(font)
690    }
691}

693/// Check lines for text overflow in a game.
694pub fn check_line_size(options: &CheckLineSizeOptions) -> anyhow::Result<CheckLineSizeIter> {
695    let font = options.load_font()?;
696    let context = CheckLineSizeContext::new(
697        font,
698        options.font_size,
699        options.screen_width,
700        options.extra_single_text_codes.clone(),
701        options.extra_text_codes.clone(),
702        options.yep_message_core,
703    );
704    let iter = CheckLineSizeIter::new(&options.data_path, context)?;
705
706    Ok(iter)
